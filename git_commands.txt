// init commands 
1. git init :- to create a new repo
2. git remote add origin <-link->
3. git remote -v (to verify remote)
4. git branch  (to check branch)
5. git branch -M main (to rename branch)
6. git push origin main


// Branch commands
1. git branch  (to check branch)
2. git branch -M main (to rename branch)
3. git checkout "branch name "  (to switch from one branch to other)
4. git checkout -b "branch name"  (to create a new branch)
5. git branch -d "branch name" (to delete branch)
